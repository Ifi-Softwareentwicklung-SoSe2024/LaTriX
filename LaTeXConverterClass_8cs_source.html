71  public static string ConvertMatrixToLaTeX<T>(Matrix<T> matrix, Alignment alignment=Alignment.center) where T : struct, IEquatable<T>, IFormattable
72  /* Die Einschränkung des Typparameters T auf ein struct, das IEquatable<T> und IFormattable implementiert, ist wegen
73  der internen Implementierung der `MathNet.Numerics.LinearAlgebra.Matrix`-Klasse zu nutzen, da nur aus solchen Typen Matrizen
74  erstellt werden können. */
75  {
76  StringBuilder latexStringBuilder = new StringBuilder(@"\left( \begin{array}{"); // Overhead, der die Matrix initialisiert
77  char alignment_char = GetAlignmentChar(alignment);
78  latexStringBuilder.Append(new string(alignment_char, matrix.ColumnCount)); // Formatspecifier der Matrix (linksbündig, rechtsbündig oder zentriert)
79  latexStringBuilder.AppendLine(@"}");
80  for (int i=0; i < matrix.RowCount; i++) // Iteration über die Reihen
81  {
82  latexStringBuilder.Append(matrix[i,0].ToString()); // Vor das erste Element darf kein & gesetzt werden, deswegen Extrabehandlung!
83  for (int j=1; j < matrix.ColumnCount; j++) // Iteration über die Spalten
84  {
85  latexStringBuilder.Append('&'); // Zwischen Zeilenelementen muss ein Trennzeichen gesetzt werden
86  latexStringBuilder.Append(matrix[i,j].ToString("G", CultureInfo.InvariantCulture)); // InvariantCulture erzwingt hier einen Dezimalpunkt!
87  }
88  latexStringBuilder.AppendLine(@"\\"); // Zeile beendet, setze LaTeX-Linebreak (\\‍) und Datei-Linebreak (\n)
89  }
90  latexStringBuilder.Append(@"\end{array} \right)"); // Matrixsuffix, Gegenstück zum Overhead
91  return latexStringBuilder.ToString();
92  }
