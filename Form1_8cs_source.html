166  {

168  {
169  // Erlaubt nur Ziffern, Steuerzeichen, einen Dezimalpunkt oder Komma und ein Minuszeichen am Anfang
170  if (!char.IsControl(e.KeyChar) && !char.IsDigit(e.KeyChar) && e.KeyChar != '.' && e.KeyChar != ',' && e.KeyChar != '-')
171  {
172  throw new Exception("Ungültige Eingabe. Bitte geben Sie eine gültige Zahl ein.");
173  }
174 
175  // Stellt sicher, dass nur ein Dezimalpunkt oder Komma erlaubt ist
176  if ((e.KeyChar == '.' || e.KeyChar == ',') && ((sender as TextBox).Text.IndexOf('.') > -1 || (sender as TextBox).Text.IndexOf(',') > -1))
177  {
178  throw new Exception("Ungültige Eingabe. Bitte geben Sie eine gültige Zahl ein.");
179  }
180 
181  // Stellt sicher, dass das Minuszeichen nur am Anfang steht
182  if (e.KeyChar == '-' && (sender as TextBox).SelectionStart != 0)
183  {
184  throw new Exception("Ungültige Eingabe. Bitte geben Sie eine gültige Zahl ein.");
185  }
186  }
187  catch (Exception ex)
188  {
189  MessageBox.Show(ex.Message, "Eingabefehler", MessageBoxButtons.OK, MessageBoxIcon.Error);
190  e.Handled = true;
