166  private void DataGridViewMatrix_KeyPress(object sender, KeyPressEventArgs e)
167  {
168  try
169  {
170  // Erlaubt nur Ziffern, Steuerzeichen, einen Dezimalpunkt oder Komma und ein Minuszeichen am Anfang
171  if (!char.IsControl(e.KeyChar) && !char.IsDigit(e.KeyChar) && e.KeyChar != '.' && e.KeyChar != ',' && e.KeyChar != '-')
172  {
173  throw new Exception("Ungültige Eingabe. Bitte geben Sie eine gültige Zahl ein.");
174  }
175 
176  // Stellt sicher, dass nur ein Dezimalpunkt oder Komma erlaubt ist
177  if ((e.KeyChar == '.' || e.KeyChar == ',') && ((sender as TextBox).Text.IndexOf('.') > -1 || (sender as TextBox).Text.IndexOf(',') > -1))
178  {
179  throw new Exception("Ungültige Eingabe. Bitte geben Sie eine gültige Zahl ein.");
180  }
181 
182  // Stellt sicher, dass das Minuszeichen nur am Anfang steht
183  if (e.KeyChar == '-' && (sender as TextBox).SelectionStart != 0)
184  {
185  throw new Exception("Ungültige Eingabe. Bitte geben Sie eine gültige Zahl ein.");
186  }
187  }
188  catch (Exception ex)
189  {
190  MessageBox.Show(ex.Message, "Eingabefehler", MessageBoxButtons.OK, MessageBoxIcon.Error);
191  e.Handled = true;
192  }

198  private void ConvertMatrixToLaTeX()
199  {
200  var matrix = Matrix<double>.Build.Dense(dataGridViewMatrix.RowCount, dataGridViewMatrix.ColumnCount);
201  for (int i = 0; i < dataGridViewMatrix.RowCount; i++)
202  {
203  for (int j = 0; j < dataGridViewMatrix.ColumnCount; j++)
204  {
205  if (double.TryParse(dataGridViewMatrix[j, i].Value?.ToString().Replace(',', '.'), NumberStyles.Any, CultureInfo.InvariantCulture, out double value))
206  {
207  matrix[i, j] = value;
208  }
209  }
210  }
